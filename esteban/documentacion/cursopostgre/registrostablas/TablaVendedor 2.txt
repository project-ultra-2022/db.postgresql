create table cantporciudad
  select concat_ws(' ', nombre, apellidopaterno,apellidomaterno) as 'nombrecompleto',s.municipio,count(*) as cantidad
  from vendedor  v
  join sucursales  s
  on v.num_sucursal=s.numsucursal  	
  group by v.nombre,s.municipio;
